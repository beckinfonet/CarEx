    ```typescript
    // Imports — mirror AuthService.ts:1-3 minus AsyncStorage (per PATTERNS: interceptor never touches storage)
    import axios, { AxiosRequestConfig } from 'axios';
    import { API_URL } from '../../constants/config';
    import { ModerationError } from '../moderation/errors';

    // Axios config extension — declaration merge so callers can pass _skipModerationInterceptor typed.
    declare module 'axios' {
      interface AxiosRequestConfig {
        _skipModerationInterceptor?: boolean;
      }
    }

    // Module-level getters registered by AuthContext on mount (D-04, D-09).
    // Use `let` (not React state) for synchronous reads inside interceptors.
    let tokenProvider: (() => string | null) | null = null;
    export function setTokenProvider(fn: () => string | null) { tokenProvider = fn; }

    let moderationRefreshListener: (() => Promise<void>) | null = null;
    export function setModerationRefreshListener(fn: () => Promise<void>) {
      moderationRefreshListener = fn;
    }

    // Shared instance — baseURL scopes every path-relative call to the carEx backend.
    // Identity Toolkit (identitytoolkit.googleapis.com) calls stay on plain `axios` per D-01.
    export const apiClient = axios.create({ baseURL: API_URL });

    // Request interceptor — unconditional Bearer per D-02.
    // Per 04-CONTEXT specifics line 248: baseURL already scopes URL-matching, no URL check needed.
    apiClient.interceptors.request.use((config) => {
      const token = tokenProvider?.();
      if (token) {
        config.headers = config.headers ?? {};
        (config.headers as any).Authorization = `Bearer ${token}`;
      }
      return config;
    });

    // Response interceptor — catches 403 account_suspended, awaits refresh, re-throws ModerationError (D-09, D-10, D-11).
    apiClient.interceptors.response.use(
      (res) => res,
      async (err) => {
        const res = err.response;
        const isAcctSuspended =
          res?.status === 403 && res?.data?.error === 'account_suspended';
        const skipFlag = (err.config as AxiosRequestConfig | undefined)?._skipModerationInterceptor === true;
        if (isAcctSuspended && !skipFlag) {
          try {
            await moderationRefreshListener?.();
          } catch (refreshErr) {
            console.error('Moderation refresh listener failed', refreshErr);
            // Swallow: we still want to surface ModerationError to the caller.
          }
          throw new ModerationError(
            'account_suspended',
            res.data?.status,
            res.data?.reasonCategory,
            res.data?.note,
            403,
          );
        }
        throw err;
      },
    );
    ```

Plan 04-04 adds the second grep-legitimate occurrence (the refreshUser call site). Third occurrence anywhere in the codebase is a bug per CONTEXT.

| T-04-01-01 | Information Disclosure | request interceptor attaching Bearer | mitigate | `baseURL: API_URL` scopes every `apiClient` request to carEx backend; Identity Toolkit calls stay on plain `axios` per D-01. Acceptance grep `identitytoolkit` in client.ts = 0. |
| T-04-01-02 | Denial of Service | response interceptor recursion (403 during refresh triggers another 403 handler) | mitigate | `_skipModerationInterceptor: true` axios config flag short-circuits the interceptor when set (D-11). | T-04-01-03 | Tampering | ModerationError constructor accepts arbitrary strings | accept | `code` union is open-ended by design (D-07) to permit new backend codes without lockstep mobile release. Error is thrown by trusted interceptor code only; no user-controlled data constructs one in callers. |

| T-04-01-06 | Repudiation | refresh listener never runs if interceptor throws before awaiting | mitigate | Interceptor awaits `moderationRefreshListener?.()` inside `try/catch` BEFORE the `throw new ModerationError(...)`. Test 5 asserts listener is called exactly once on 403 account_suspended. |
