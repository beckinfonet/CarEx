    The suspend body implements the full D-18/D-19/D-20/D-24/D-27 pattern. It uses the TWO-STEP transaction pattern (insert audit → read _id → update user with lastActionId) — D-specifics warns "do NOT optimize into a single $set."

    Expected service.js structure (extend the existing scaffold — do not rewrite imports that are already there):

    ```javascript
    // src/moderation/service.js
    const mongoose = require('mongoose');
    const User = require('../models/User');
    const AdminUser = require('../models/AdminUser');
    const ModerationAction = require('../models/ModerationAction');
    const { resolveRestrictedFeatures } = require('./capabilities');

    // NOTE: We bypass actions.writeAction() inside transactions because writeAction uses
    // ModerationAction.create(singleDoc) which cannot accept { session }. The array form
    // (ModerationAction.create([doc], { session })) is required by Mongoose to pass options.
    // writeAction() remains the canonical path for non-transactional audit writes; Phase 2
    // handlers ALWAYS write inside a transaction, so they call create() directly with session.
    // Audit-ledger integrity is preserved because every transactional path here follows the
    // same shape (ModerationAction.create within withTransaction).

    // --- suspend ------------------------------------------------------------
    async function suspend({ adminUid, adminEmail, targetUid, severity, reasonCategory, note }) {
      if (!adminUid || !adminEmail || !targetUid || !severity || !reasonCategory) {
        throw new Error('suspend: adminUid, adminEmail, targetUid, severity, reasonCategory are required');
      }

      // Pre-transaction read: confirm target exists + detect re-suspend-same-severity
      // idempotency violation (D-20). Doing this outside the transaction is OK — the
      // transaction re-reads before mutating; this is just a fast-path 400.
      const target = await User.findOne({ firebaseUid: targetUid }).lean();
      if (!target) throw new Error('target_not_found');
      if (
        target.moderationStatus &&
        target.moderationStatus.state === severity &&
        target.moderationStatus.severity === severity
      ) {
        throw new Error('already_at_severity');
      }

      const restrictedFeatures = resolveRestrictedFeatures(severity);
      const setAt = new Date();

      const session = await mongoose.startSession();
      let insertedAction;
      let newModerationStatus;
      try {
        await session.withTransaction(async () => {
          // 1. Insert audit row FIRST (needed for lastActionId back-link per D-18).
          const [action] = await ModerationAction.create([{
            targetUid,
            adminUid,
            adminEmail,
            action: 'suspend',
            severity,
            reasonCategory,
            note: note ?? null,
          }], { session });
          insertedAction = action;

          // 2. Last-admin guard (D-27, D-28) — suspend only. Inside the transaction so the
          //    read + write are isolated as a unit. Query active admins: join AdminUser.email
          //    to User to count Users whose moderationStatus.state is 'active'.
          const adminEmails = await AdminUser.distinct('email', {}, { session });
          const activeAdminCount = await User.countDocuments({
            email: { $in: adminEmails },
            'moderationStatus.state': 'active',
          }).session(session);

          // If the TARGET is an active admin AND would-be count after this suspend drops to 0,
          // reject. We compare pre-mutation: if target is currently counted (state==='active'
          // and email is in adminEmails), activeAdminCount includes them; after mutation it
          // would be count - 1. Protect when (count - 1) <= 0 AND target is currently counted.
          const targetIsActiveAdmin =
            adminEmails.includes(target.email) &&
            (target.moderationStatus ? target.moderationStatus.state === 'active' : true);
          if (targetIsActiveAdmin && activeAdminCount - 1 <= 0) {
            throw new Error('last_admin_protected');
          }

          // 3. Update User.moderationStatus with the new subdoc + back-link to audit row.
          newModerationStatus = {
            state: severity,
            severity,
            reasonCategory,
            note: note ?? null,
            setByAdminUid: adminUid,
            setAt,
            restrictedFeatures,
            lastActionId: action._id,
          };
          const updated = await User.updateOne(
            { firebaseUid: targetUid },
            { $set: { moderationStatus: newModerationStatus } },
            { session }
          );
          if (updated.matchedCount !== 1) {
            throw new Error('target_not_found');
          }
        });
      } finally {
        await session.endSession();
      }

      return {
        ok: true,
        user: { moderationStatus: newModerationStatus },
        action: {
          _id: insertedAction._id.toString(),
          action: insertedAction.action,
          createdAt: insertedAction.createdAt,
        },
      };
    }

    // --- unsuspend ----------------------------------------------------------
    async function unsuspend({ adminUid, adminEmail, targetUid, note }) {
      if (!adminUid || !adminEmail || !targetUid) {
        throw new Error('unsuspend: adminUid, adminEmail, targetUid are required');
      }

      const target = await User.findOne({ firebaseUid: targetUid }).lean();
      if (!target) throw new Error('target_not_found');

      const currentState = target.moderationStatus ? target.moderationStatus.state : 'active';
      if (currentState === 'active') {
        throw new Error('not_suspended');
      }

      const setAt = new Date();
      const session = await mongoose.startSession();
      let insertedAction;
      let newModerationStatus;
      try {
        await session.withTransaction(async () => {
          const [action] = await ModerationAction.create([{
            targetUid,
            adminUid,
            adminEmail,
            action: 'unsuspend',
            severity: 'none',
            reasonCategory: null,
            note: note ?? null,
          }], { session });
          insertedAction = action;

          newModerationStatus = {
            state: 'active',
            severity: 'none',
            reasonCategory: null,
            note: null,
            setByAdminUid: adminUid,
            setAt,
            restrictedFeatures: [],
            lastActionId: action._id,
          };
          const updated = await User.updateOne(
            { firebaseUid: targetUid },
            { $set: { moderationStatus: newModerationStatus } },
            { session }
          );
          if (updated.matchedCount !== 1) throw new Error('target_not_found');
        });
      } finally {
        await session.endSession();
      }

      return {
        ok: true,
        user: { moderationStatus: newModerationStatus },
        action: {
          _id: insertedAction._id.toString(),
          action: insertedAction.action,
          createdAt: insertedAction.createdAt,
        },
      };
    }

    // --- stubs still owned by Plans 02-04 and 02-05 -------------------------
    async function revokeRole(/* { adminUid, adminEmail, targetUid, role, reasonCategory, note } */) {
      throw new Error('revokeRole: not implemented (Plan 02-04)');
    }
    async function deleteProviderProfile(/* { ... } */) {
      throw new Error('deleteProviderProfile: not implemented (Plan 02-05)');
    }
    async function editProfile(/* { ... } */) {
      throw new Error('editProfile: not implemented (Plan 02-05)');
    }

    module.exports = { suspend, unsuspend, revokeRole, deleteProviderProfile, editProfile };
    ```

    ```javascript
    // src/moderation/router.js
    const express = require('express');
    const service = require('./service');
    const { denySelfModeration } = require('./denySelfModeration');
    const { dispatchSchema, unsuspendSchema } = require('./schemas');

    const router = express.Router();

    // Errors the service throws that translate to user-facing 400s (Phase 2). Anything else
    // is an internal_error (500).
    const KNOWN_USER_ERRORS = new Set([
      'already_at_severity',
      'not_suspended',
      'last_admin_protected',
      'target_not_found',
      'role_not_assigned',       // Plans 02-04, 02-05
      'invalid_field',           // Plan 02-05 (edit-profile)
      'no_changes',              // Plan 02-05 (edit-profile)
      'invalid_role_for_delete', // Plan 02-05
    ]);

    function handleServiceError(err, res, tag) {
      if (KNOWN_USER_ERRORS.has(err.message)) {
        return res.status(400).json({ error: err.message });
      }
      // eslint-disable-next-line no-console
      console.error(`[moderation] ${tag} error:`, err);
      return res.status(500).json({ error: 'internal_error', message: err.message });
    }

    // Scaffold route from Phase 1 — unchanged.
    router.get('/ping', (req, res) => {
      res.json({ ok: true });
    });

    // POST /:targetUid — dispatch on body.action. Covers suspend (Plan 02-03) and
    // revoke_role (Plan 02-04). The discriminatedUnion in dispatchSchema routes to the
    // correct per-action schema before we ever see body.
    router.post('/:targetUid', denySelfModeration, async (req, res) => {
      const parsed = dispatchSchema.safeParse(req.body);
      if (!parsed.success) {
        return res.status(400).json({ error: 'invalid_payload', issues: parsed.error.issues });
      }
      try {
        if (parsed.data.action === 'suspend') {
          const result = await service.suspend({
            adminUid: req.admin.uid,
            adminEmail: req.admin.email,
            targetUid: req.params.targetUid,
            severity: parsed.data.severity,
            reasonCategory: parsed.data.reasonCategory,
            note: parsed.data.note,
          });
          return res.json(result);
        }
        if (parsed.data.action === 'revoke_role') {
          // Plan 02-04 replaces this with service.revokeRole(...).
          return res.status(501).json({ error: 'not_implemented', message: 'revoke_role lands in Plan 02-04' });
        }
        // Unreachable — dispatchSchema enforces the enum.
        return res.status(400).json({ error: 'invalid_payload' });
      } catch (err) {
        return handleServiceError(err, res, 'suspend/dispatch');
      }
    });

    // PATCH /:targetUid/unsuspend — dedicated route (not in POST dispatch per D-01).
    router.patch('/:targetUid/unsuspend', denySelfModeration, async (req, res) => {
      const parsed = unsuspendSchema.safeParse(req.body || {});
      if (!parsed.success) {
        return res.status(400).json({ error: 'invalid_payload', issues: parsed.error.issues });
      }
      try {
        const result = await service.unsuspend({
          adminUid: req.admin.uid,
          adminEmail: req.admin.email,
          targetUid: req.params.targetUid,
          note: parsed.data.note,
        });
        return res.json(result);
      } catch (err) {
        return handleServiceError(err, res, 'unsuspend');
      }
    });

    module.exports = router;
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| client body → dispatchSchema | Zod strict parse is the gate; unknown fields, wrong severities rejected before service runs |
| service.suspend → transaction | audit-row insert + last-admin guard + user.moderationStatus update must commit atomically or fully roll back |
| transaction commit → audit ledger | append-only guarantee at the schema level (Phase 1 D-17) means no post-commit tamper |

## STRIDE Threat Register

| Threat ID | Category | Component | Severity | Disposition | Mitigation |
|-----------|----------|-----------|----------|-------------|------------|
| T-02-03-01 | Tampering | TOCTOU on last-admin guard (Pitfall 2) | high | mitigate | Last-admin count query runs inside `session.withTransaction()` with `.session(session)` — read/write isolation across guard + mutation. If a concurrent admin suspend would drop count to 0, both transactions cannot commit. |
| T-02-03-02 | Integrity | audit row committed but user update fails | high | mitigate | Two-step pattern (audit → user) inside one `withTransaction()`. If the user update throws, the transaction aborts and the audit row is rolled back — no orphan audit entries. |
| T-02-03-03 | Integrity | lastActionId back-link drift | medium | mitigate | Audit row is created FIRST inside the transaction; its `_id` is captured and embedded in the user update within the same session. No concurrent suspend on the same user can interleave. |
| T-02-03-04 | Elevation of Privilege | self-suspend bypassing middleware | high | mitigate | `denySelfModeration` is applied per-route on both POST and PATCH. Backend is authoritative — D-26 explicitly calls out that UI guard is defense-in-depth. |
| T-02-03-05 | Elevation of Privilege | admin lockout via self-targeted suspend (last admin) | high | mitigate | Self-moderation is blocked by denySelfModeration BEFORE the last-admin query. Even if bypassed, the last-admin guard runs inside the transaction. Two-tier defense per D-26 + D-28. |
| T-02-03-06 | Repudiation | re-suspend silently overwrites audit history | medium | mitigate | ModerationAction is append-only (Phase 1 D-17) — re-suspend writes a NEW row; the prior row is schema-protected from mutation or delete. Test 2 asserts both rows exist after re-suspend. |
| T-02-03-07 | Information Disclosure | 500 leaks err.message with internal detail | low | accept | `handleServiceError` returns `{ error: 'internal_error', message: err.message }` to admins only. Admins are authenticated + authorized; leak blast radius is contained. Pattern matches Phase 1 D-10. |
| T-02-03-08 | DoS | transaction deadlock on high-contention target | low | accept | `session.withTransaction()` auto-retries transient transaction errors (Mongoose default). Worst case: handler throws after retries — 500 to the admin, no state change. Acceptable at expected scale. |
| T-02-03-09 | Tampering | rate-limit bypass (not yet wired) | high | deferred-to-02-06 | Plan 02-06 mounts the rate limiter + validates 30/15min envelope.
