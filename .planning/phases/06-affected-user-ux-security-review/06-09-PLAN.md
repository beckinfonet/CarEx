    ```typescript
    import fs from 'fs';
    import path from 'path';

    // Files scanned by QUAL-01 literal-string audit (Phase 6).
    // A "user-facing literal" is a JSX Text child that is a string literal (not wrapped in {t.*} or {variable}).
    // Brand/proper-noun literals are allowlisted below.
    const SCAN_FILES = [
      'src/components/moderation/UserStatusBanner.tsx',
      'src/components/moderation/FeatureGateOverlay.tsx',
      'src/components/moderation/GatedScreenWrapper.tsx',
    ];

    // Tokens that are legitimate to appear as a Text child without translation.
    // Keep tight; expanding this list without justification defeats the audit.
    const ALLOWLIST = new Set<string>([
      // (none for now — all strings in the new components are expected to come from t.*)
    ]);

    // Matches `<Text ...>LITERAL</Text>` where LITERAL contains letters and is NOT
    // purely whitespace/punctuation. `<Text>{...}</Text>` forms are naturally excluded
    // because the captured group would start with `{`, not a letter.
    const TEXT_LITERAL_REGEX = /<Text[^>]*>\s*([A-Za-zА-Яа-я][^<{}]*?)\s*<\/Text>/g;

    function scanFile(absPath: string): Array<{ line: number; literal: string }> {
      const src = fs.readFileSync(absPath, 'utf8');
      const lines = src.split('\n');
      const offenders: Array<{ line: number; literal: string }> = [];

      for (let i = 0; i < lines.length; i++) {
        const line = lines[i];
        TEXT_LITERAL_REGEX.lastIndex = 0;
        let m: RegExpExecArray | null;
        while ((m = TEXT_LITERAL_REGEX.exec(line)) !== null) {
          const literal = m[1].trim();
          if (literal.length === 0) continue;
          if (ALLOWLIST.has(literal)) continue;
          offenders.push({ line: i + 1, literal });
        }
      }
      return offenders;
    }

    describe('QUAL-01: moderation components — no untranslated literals', () => {
      for (const relPath of SCAN_FILES) {
        test(`${relPath} contains no user-facing <Text> literals outside t.*`, () => {
          const absPath = path.resolve(__dirname, '..', relPath);
          expect(fs.existsSync(absPath)).toBe(true);
          const offenders = scanFile(absPath);
          if (offenders.length > 0) {
            const diag = offenders
              .map(o => `  line ${o.line}: "${o.literal}"`)
              .join('\n');
            throw new Error(
              `Found ${offenders.length} untranslated <Text> literal(s) in ${relPath}:\n${diag}\n` +
              `Wrap them in t.* via useLanguage(), or add to ALLOWLIST with justification.`
            );
          }
          expect(offenders).toEqual([]);
        });
      }
    });
    ```

    **Important design notes:**
    - Use fs (synchronous) + regex — no AST parser (ts-morph / babel-traverse) because that requires a new dev dep (forbidden by CLAUDE.md discipline)
    - The regex captures JSX Text children that start with a letter (Latin or Cyrillic). `<Text>{t.foo}</Text>` starts with `{`, not a letter → excluded. Pure punctuation/whitespace children are also naturally excluded
    - Multiline Text bodies (rare in these components) are partially covered — the regex matches within a line. If a multiline literal slips through, the scanner under-reports; that's acceptable for this audit level. Add an AST parser in a future phase if this becomes a real bypass
    - If a legitimate literal must stay (e.g., a phone number format constant), add to ALLOWLIST with a code-comment justifying the entry. Initially empty
    - Test file exits 0 (PASS) against the Phase 6 components because UserStatusBanner, FeatureGateOverlay, GatedScreenWrapper all use `t.*` for every user-visible Text child by Plans 06-03/04/05 contract
