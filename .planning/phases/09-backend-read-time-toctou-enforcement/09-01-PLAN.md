```
exports: {
  refundAndThrow(stripe, paymentIntentId, errorBody) -> Promise<never> (always throws)
    - Calls stripe.refunds.create({ payment_intent: paymentIntentId }, { idempotencyKey: `refund-${paymentIntentId}` }) FIRST
    - On success: refundId = result.id, refundFailed = false
    - On failure: refundFailed = true, refundId = null, console.error logged
    - Discriminator on errorBody.error:
        'listing_not_available' -> throw new ListingNotAvailableError({ ...errorBody, refundId, refundFailed })
        anything else            -> throw ProviderSuspendedError(errorBody.error) with Object.assign of enriched body
  ListingNotAvailableError(body) extends Error
    - this.name = 'ListingNotAvailableError'
    - Object.assign(this, body) so .listingStatus / .reasonCategory / .banner / .refundId / .refundFailed are readable
    - super(body.error) so .message === 'listing_not_available'
}
```

```
const AdminUser = require('../models/AdminUser');

async function lookupAdminIfPresent(req, res, next) {
```
