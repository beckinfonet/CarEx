```js
const admin = require('firebase-admin');

let initialized = false;

/**
 * Lazy-initialize firebase-admin once. Uses FIREBASE_SERVICE_ACCOUNT_JSON env var
 * containing the stringified JSON from the Google Cloud console (D-07).
 * Throws a helpful error if the env var is missing or malformed so misconfig is loud.
 */
function ensureInitialized() {
  if (initialized) return admin;
  const raw = process.env.FIREBASE_SERVICE_ACCOUNT_JSON;
  if (!raw) {
    throw new Error('[firebaseAdmin] FIREBASE_SERVICE_ACCOUNT_JSON env var is not set. See .env.example.');
  }
  let parsed;
  try {
    parsed = JSON.parse(raw);
  } catch (err) {
    throw new Error(`[firebaseAdmin] FIREBASE_SERVICE_ACCOUNT_JSON is not valid JSON: ${err.message}`);
  }
  admin.initializeApp({ credential: admin.credential.cert(parsed) });
  initialized = true;
  return admin;
}

module.exports = { admin, ensureInitialized };
```

Target `src/security/verifyIdToken.js`:
```js
const { ensureInitialized } = require('./firebaseAdmin');

/**
 * Express middleware. Parses `Authorization: Bearer <idToken>`, verifies via firebase-admin,
 * attaches req.auth = { uid, email, claims }. Returns 401 with the standard body on failure.
 *
 * 401 shape per D-10:
 *   { error: 'unauthenticated', message: 'Missing or invalid idToken' }
 */
async function verifyIdToken(req, res, next) {
  const header = req.header('authorization') || req.header('Authorization') || '';
  const match = header.match(/^Bearer (.+)$/);
  if (!match) {
    return res.status(401).json({ error: 'unauthenticated', message: 'Missing or invalid idToken' });
  }
  try {
    const admin = ensureInitialized();
    const decoded = await admin.auth().verifyIdToken(match[1], true);
    req.auth = { uid: decoded.uid, email: decoded.email, claims: decoded };
    return next();
  } catch (err) {
    return res.status(401).json({ error: 'unauthenticated', message: 'Missing or invalid idToken' });
  }
}

module.exports = { verifyIdToken };
```

Target `src/security/requireAdmin.js`:
```js
const AdminUser = require('../models/AdminUser');

/**
 * Express middleware. Requires verifyIdToken upstream (reads req.auth.email).
 * Looks up AdminUser by email (case-insensitive), attaches req.admin = { role, email }.
 *
 * 403 shape per D-10:
 *   { error: 'unauthorized', message: 'Admin access required' }
 */
async function requireAdmin(req, res, next) {
  if (!req.auth || !req.auth.email) {
    return res.status(403).json({ error: 'unauthorized', message: 'Admin access required' });
  }
  const admin = await AdminUser.findOne({ email: req.auth.email.toLowerCase() }).lean();
  if (!admin) {
    return res.status(403).json({ error: 'unauthorized', message: 'Admin access required' });
  }
  req.admin = { role: admin.role, email: admin.email };
  return next();
}

module.exports = { requireAdmin };
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Client → /api/admin/moderation/* | THE trust boundary for all Phase 2 moderation routes; crossing requires a valid Firebase idToken signed by Google's rotating public keys. |
| verifyIdToken → AdminUser collection | Authorization (not authentication) decision; reads AdminUser by email from the verified token. |
| Railway env vars → firebase-admin init | Service account JSON lives in FIREBASE_SERVICE_ACCOUNT_JSON; source tree never sees it. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-05-01 | Spoofing | Caller impersonating an admin via forged UID in request body | mitigate | verifyIdToken middleware discards any body/query param claiming identity and uses ONLY the Firebase-verified uid/email from the Authorization: Bearer token. Addresses Pitfall #1 directly. Test "valid idToken but email not admin → 403" proves the policy. |
| T-05-02 | Tampering | Token signature tampered | mitigate | firebase-admin.auth().verifyIdToken validates signature against Google's rotating JWKs; any mutation fails verification → 401. The `true` second arg enables checkRevoked so revoked-user tokens also fail. |
| T-05-03 | Elevation of privilege | Non-admin with a valid user idToken hits moderation endpoint | mitigate | requireAdmin does a server-side AdminUser.findOne lookup against the verified email — client cannot bypass by claiming role. Test "valid idToken but email not admin → 403" asserts this. |
| T-05-04 | Repudiation | Absence of idToken verification on admin action allows anonymous admin ops | mitigate | Every route under /api/admin/moderation goes through verifyIdToken FIRST, then requireAdmin. The GET /ping scaffold exercises this path end-to-end so the test becomes the living proof that the chain cannot be bypassed. |
| T-05-05 | Info disclosure | Error responses differ on "no admin exists" vs "wrong admin" | mitigate | Both non-admin and missing-email cases return the same 403 body { error: 'unauthorized', message: 'Admin access required' }. No enumeration via error variance. |
| T-05-06 | Elevation of privilege | Service account JSON accidentally committed to git | mitigate | .env.example documents only the var NAME; actual JSON lives in Railway env and the .env file is already gitignored (verified via `cat .gitignore | grep -c .env`). D-08 is explicit. |
| T-05-07 | Denial of service | firebase-admin init fails at startup → every admin call 500s | mitigate | ensureInitialized throws a clear error on first call so misconfig is loud, not silent. Operator gets a log line they can act on. |
| T-05-08 | Elevation of privilege | Legacy /api/admin/requests still spoofable | accept | D-06 explicitly defers legacy migration; tracked as follow-up. Phase 2 Summary will call out the remaining exposure. The Phase 1 improvement is that NEW moderation routes are not spoofable — the surface area does not grow. |
| T-05-09 | Info disclosure | verifyIdToken decodes and attaches full claims to req.auth.claims | accept | Claims are attached to the server-side request object, not returned in responses. Future endpoints may need specific claims (e.g., email_verified). Attaching them is cheaper than re-decoding. |
| T-05-10 | Tampering | Email-case mismatch between Firebase identity and AdminUser seed | mitigate | requireAdmin lowercases req.auth.email before lookup; seedSuperAdmin already lowercases. Test "email case-insensitive match → 200" asserts this consistency. |
</threat_model>
