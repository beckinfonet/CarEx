```js
      bodyKey: 'moderation.permanently_banned.body',

/**
 * Resolve the list of restricted feature tokens for a given moderation state.
 * Used by Phase 2's moderation service when writing user.moderationStatus.restrictedFeatures (D-12).
 *
 * @param {'active'|'feature_limited'|'blocked_with_review'|'permanently_banned'} state
 * @returns {string[]} restricted feature tokens; empty array if state is 'active';
 *                    the sentinel ['all_writes'] for blocked_with_review / permanently_banned
 */
function resolveRestrictedFeatures(state) {
  const entry = STATUS_POLICY[state];
  if (!entry) throw new Error(`Unknown moderation state: ${state}`);
  if (entry.capabilities === 'all') return [];
  if (entry.capabilities.blocked === 'all_writes') return ['all_writes'];
  if (Array.isArray(entry.capabilities.blocked)) return [...entry.capabilities.blocked];
  throw new Error(`Malformed STATUS_POLICY entry for state: ${state}`);
}

module.exports = { STATUS_POLICY, resolveRestrictedFeatures };
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| capabilities.js → future Phase 2/3 enforcement code | STATUS_POLICY is the authoritative policy; any enforcement that reads raw state strings instead of consulting STATUS_POLICY creates drift (Pitfall #11 — gating sprawl). |
| STATUS_POLICY keys ↔ User.moderationStatus.state enum | Drift here would cause resolveRestrictedFeatures to silently return wrong values for real DB states. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-04-01 | Tampering | Drift between STATUS_POLICY keys and User state enum | mitigate | Test "STATUS_POLICY keys match User.moderationStatus.state enum" fails CI if anyone adds a state to one without the other. |
| T-04-02 | Elevation of privilege | Unknown state → resolveRestrictedFeatures returns [] (no restrictions) | mitigate | resolveRestrictedFeatures THROWS on unknown state rather than returning empty. Phase 2 enforcement will therefore fail closed (loud error) rather than fail open (silent permit). Test "throws on unknown state" asserts this. |
| T-04-03 | Info disclosure | support@carexmarket.com baked into source | accept | This is a public support address (documented in the appeal flow). Not a secret. |
| T-04-04 | Tampering | A subsequent plan adds new capability tokens without updating the test | accept | The test enumerates the 7 tokens exactly. Adding an 8th requires a test edit — that's the intended friction. Phase 1's job is to lock the Phase 1 contract. |
| T-04-05 | Repudiation | STATUS_POLICY used to grant rights not matched by server-side enforcement | mitigate | Per Pitfall #11, mobile UI gating is UX only — real enforcement lives in Phase 3 middleware. Phase 1's capability map is shared contract; Phase 3 owns enforcement. Separation of concerns is the mitigation. |
</threat_model>
